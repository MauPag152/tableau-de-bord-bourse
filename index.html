packages = ["yfinance", "pandas", "numpy", "matplotlib", "scikit-learn"] import yfinance as yf import pandas as pd import numpy as np import matplotlib.pyplot as plt from sklearn.linear_model import LinearRegression from js import document def run_regression(): try: # 1. Collecte des Données (AAPL - 1 an) ticker = "AAPL" data = yf.download(ticker, period="1y") # Gestion des en-têtes doubles yfinance if isinstance(data.columns, pd.MultiIndex): df = data['Close'][ticker].to_frame() else: df = data['Close'].to_frame() df = df.dropna().tail(252) y = df.values.flatten().astype(float) X = np.arange(len(y)).reshape(-1, 1) # 2. Paramètres Statistiques model = LinearRegression() model.fit(X, y) y_pred = model.predict(X) # Calcul de l'erreur type pour les canaux std_error = np.std(y - y_pred) # 3. Graphique Ultra-Large (25, 8) fig, ax = plt.subplots(figsize=(25, 8), facecolor='white') # Courbe des prix bleu foncé ax.plot(df.index, y, color='#00008B', linewidth=2, label=f'Prix {ticker}') # Ligne centrale noire et canaux ax.plot(df.index, y_pred, color='black', linewidth=1.5, label='Tendance Centrale') # Limites et Shading (Zones de couleurs) ax.plot(df.index, y_pred + 2*std_error, color='red', linestyle='--', alpha=0.6, label='+2 Std Dev') ax.plot(df.index, y_pred + 1*std_error, color='orange', linestyle='--', alpha=0.6) ax.plot(df.index, y_pred - 1*std_error, color='blue', linestyle='--', alpha=0.6) ax.plot(df.index, y_pred - 2*std_error, color='green', linestyle='--', alpha=0.6) ax.fill_between(df.index, y_pred - 2*std_error, y_pred + 2*std_error, color='gray', alpha=0.05) # Point Final et Étiquette last_price = float(y[-1]) ax.scatter(df.index[-1], last_price, color='red', s=100, zorder=5) ax.annotate(f"Dernier: {last_price:.2f} $", xy=(df.index[-1], last_price), xytext=(15, 0), textcoords='offset points', bbox=dict(boxstyle='round,pad=0.5', fc='white', ec='red', lw=2), fontsize=12, fontweight='bold', color='black') ax.set_title(f"ANALYSE DE RÉGRESSION LINÉAIRE - {ticker}", fontsize=18, fontweight='bold', pad=20) ax.grid(True, alpha=0.2) ax.legend(loc='upper left') # Affichage document.getElementById("status").innerHTML = "ANALYSE TERMINÉE" document.getElementById("status").classList.remove("animate-pulse") document.getElementById("plot-output").innerHTML = "" display(fig, target="plot-output") except Exception as e: document.getElementById("status").innerHTML = "ERREUR" document.getElementById("plot-output").innerHTML = f"Erreur technique : {str(e)}" run_regression()
